```sh

```

  export LDFLAGS="-L/usr/local/opt/openblas/lib"

For compilers to find lapack you may need to set:
  export LDFLAGS="-L/usr/local/opt/lapack/lib"
  export CPPFLAGS="-I/usr/local/opt/lapack/include"
